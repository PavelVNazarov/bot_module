Если у Вас еще нет суперпользователя, создайте его с помощью команды:

python manage.py createsuperuser


структура проекта

myproject/
│
├── bot/                          #
│   └── migrations/               # описания изменений формы базы данных
│   └── templates/                # Шаблоны для рендеринга страниц
│          └──bot/
│               └──admin_dashboard.html            # шаблон админки
│   ├── __init__.py               # Пустой файл для определения каталога как пакета Python
│   ├── admin.py                  # Регистрация моделей в административной панели
│   ├── apps.py                   # Конфигурация приложения
│   ├── binannce.py                #
│   ├── models.py                 # Модели данных (Книги и Читатели)
│   ├── views.py                  # Логика представлений (отрисовка страниц)
│   ├── urls.py                   # URL- маршруты для приложения "Библиотека"
│   ├── bot.py                    #
│   └── tests.py                  #
│
├── myproject/                    # Основной каталог проекта
│   ├── __init__.py               # Пустой файл для определения каталога как пакета Python
│   ├── settings.py               # Настройки проекта
│   ├── urls.py                   # Основные URL- маршруты проекта
│   ├── wsgi.py                   # Точка входа для WSGI-сервера
│   └── asgi.py                   # Точка входа для ASGI-сервера (если использовать WebSocket)
│
│
├── manage.py                      # Скрипт управления проектом Django
├── db.sqlite3                     # база данных о книгах, читателях и их связях
└── requirements.txt               # Список необходимых пакетов для проекта
